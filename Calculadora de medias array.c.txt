#include <stdio.h>

int main () {
	
	float n1[3], n2[3], n3[3], n4[3], medias[3], soma = 0, media[3];
	int i;
	
	//Entrada das notas e cálculo das medias
	
	for (i = 0; i < 3; i++){
		printf("Digite a nota %d: ", i);
		scanf("%f%f%f%f", &n1[i], &n2[i], &n3[i], &n4[i]);
		
		media[i] = (n1[i] * 2 + n2[i] * 3 + n3[i] * 4 + n4[i] * 1) / 10;
		
		
	}
	
	for(i = 0; i < 3; i++){
			printf("Media %d = %.1f\t", i, media[i]);
		}

	
	//Aprovado ou reprovado
	printf("\n");
	printf("\nN1\tN2\tN3\tN4\tMedia\tResultado\n");	
	for (i = 0; i < 3; i++) {
	
		if(media[i] >= 5) 
			printf("%.1lf\t%.1lf\t%.1lf\t%.1lf\t%.1lf\tAprovado! \n", n1[i], n2[i], n3[i], n4[i], media[i]);
			
		else if(media[i] < 5){
			printf("%.1lf\t%.1lf\t%.1lf\t%.1lf\t%.1lf\tReprovado!\n", n1[i], n2[i], n3[i], n4[i], media[i]);
			
			
		}
		
	}
	
	return 0;

	}
